    def compute_semantic_entropy(self, responses: List[str]) -> float:
        """
        Calculate semantic entropy from multiple responses.
        Lower entropy = more consistent (lower hallucination risk)
        Higher entropy = more diverse/inconsistent (higher hallucination risk)
        """
        if len(responses) < 2:
            return 0.0

        # Step 1: Generate embeddings
        embeddings = self.model.encode(responses, convert_to_numpy=True)

        semantic_entropy = entropy(cluster_probs, base=2)
        return round(semantic_entropy, 4)

Pythonimport time
import random

def call_with_retry(self, prompt: str, temperature: float, max_retries: int = 5):
    """Manual retry with exponential backoff + jitter"""
    for attempt in range(max_retries):
        try:
            return self._call_openai_api(prompt, temperature)
        except (openai.error.RateLimitError, openai.error.APIError) as e:
            if attempt == max_retries - 1:
                raise e
            wait_time = (2 ** attempt) + random.uniform(0, 1)  # Exponential + jitter
            print(f"Rate limit hit. Waiting {wait_time:.1f}s before retry {attempt + 2}...")
            time.sleep(wait_time)

Python"""
Sovereign Node 9010 v3.6 — Dataset Integrator
Integrates external deception/hallucination datasets (e.g., Mu-SHROOM) into Facts Registry
Follows new coding standards: error handling, validation, logging, type hints
"""

import json
import logging
from dataclasses import dataclass
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, List, Any, Optional

from facts_registry import FactsRegistry

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger("DatasetIntegrator")

@dataclass
class DatasetEntry:
    """Validated dataset entry for Facts Registry"""
    dataset_name: str
    sample_id: str
    text: str
    label: str  # hallucination / deception / truthful
    entropy: float
    metadata: Dict[str, Any]

class DatasetIntegrator:
    def __init__(self):
        self.registry = FactsRegistry()
        self.valid_datasets = {"Mu-SHROOM", "CCHall", "HaluEval-QA"}

    def validate_entry(self, entry: Dict[str, Any]) -> DatasetEntry:
        """Input validation per new coding standards"""
        if not isinstance(entry, dict):
            raise ValueError("Entry must be a dictionary")
        required = {"sample_id", "text", "label"}
        if not required.issubset(entry.keys()):
            raise ValueError(f"Missing required keys: {required - set(entry.keys())}")
        if len(entry["text"]) < 10:
            raise ValueError("Text too short for meaningful analysis")
        if entry["label"] not in {"hallucination", "deception", "truthful"}:
            raise ValueError("Invalid label")
        entropy = float(entry.get("entropy", 0.0))
        if not 0.0 <= entropy <= 8.0:
            raise ValueError("Entropy must be between 0.0 and 8.0")
        return DatasetEntry(
            dataset_name=entry.get("dataset_name", "unknown"),
            sample_id=entry["sample_id"],
            text=entry["text"],
            label=entry["label"],
            entropy=entropy,
            metadata=entry.get("metadata", {})
        )

    def integrate_mu_shroom_sample(self, sample: Dict[str, Any]) -> str:
        """Integrate single Mu-SHROOM sample into Facts Registry"""
        validated = self.validate_entry(sample)
        
        # Tau ceiling check
        if validated.entropy > 4.5:
            logger.warning(f"High entropy sample {validated.sample_id} — potential synthetic")
        
        fact_uuid = self.registry.add_fact(
            category=f"DATASET_{validated.dataset_name.upper()}",
            statement=validated.text[:500],
            folder_mapping="04_Validation",
            initial_cvs=validated.entropy / 8.0,  # Normalized 0-1
            deception_flag=1 if validated.label in {"hallucination", "deception"} else 0
        )
        
        logger.info(f"Integrated Mu-SHROOM sample {validated.sample_id} → Fact {fact_uuid}")
        return fact_uuid

    def integrate_batch(self, samples: List[Dict[str, Any]]) -> List[str]:
        """Batch integration with full validation"""
        fact_uuids = []
        for sample in samples:
            try:
                uuid = self.integrate_mu_shroom_sample(sample)
                fact_uuids.append(uuid)
            except Exception as e:
                logger.error(f"Failed to integrate sample: {e}")
        return fact_uuids
